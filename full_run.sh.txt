#!/bin/bash

echo "========================================"
echo "🚀 BANA 7075 - Pneumonia Detection CNN"
echo "========================================"
echo ""

# Check if data directory exists
if [ ! -d "data" ]; then
    echo "❌ Error: 'data/' directory not found!"
    echo "Please download the Chest X-Ray Pneumonia dataset from Kaggle"
    echo "and place it under the 'data/' folder as described in README.md"
    echo ""
    exit 1
fi

echo "✅ Data directory found. Starting training..."
echo ""

# Run training
python CNN_training.py

# Check if training was successful (model file exists)
if [ -f "xray_cnn_model.keras" ]; then
    echo ""
    echo "✅ Training completed successfully!"
    echo "Now running evaluation on test set..."
    echo ""
    
    # Run testing
    python CNN_testing.py
else
    echo ""
    echo "❌ Training failed or model was not saved."
    echo "Please check the errors above."
    exit 1
fi

echo ""
echo "========================================"
echo "🎉 Full run completed!"
echo "You can now run the Streamlit UI with:"
echo "   streamlit run ui.py"
echo "========================================"
